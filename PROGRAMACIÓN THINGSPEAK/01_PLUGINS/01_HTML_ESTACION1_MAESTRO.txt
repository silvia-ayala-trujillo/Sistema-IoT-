function cambiarAutomatico(estado){
  if(controlEnProceso)return;

  if(estado){

  }else{

  }
}

function cambiarManual(estado){
  if(controlEnProceso)return;

  }else{
    enviarControlSeguro(0,0,"Modo manual OFF");
  }
}

function enviarControlSeguro(autoVal,manualVal,texto){
  controlEnProceso=true;
  bloquearTodo(true);

  ultimoModoAuto=autoVal;
  ultimoManual=manualVal;

  var tActual=ultimoPunto(puntosCampoCacheTemp||[]);

  if(manualVal===1){
    estadoEstimadoCalefactor=1;
  }else if(autoVal===0&&manualVal===0){
    estadoEstimadoCalefactor=0;
  }else if(autoVal===1&&tActual){
    if(tActual.valor<=25){estadoEstimadoCalefactor=1;}
    else if(tActual.valor>=40){estadoEstimadoCalefactor=0;}
  }

  actualizarSwitches();
  actualizarEstadoVisual({
    valor:estadoEstimadoCalefactor,
    tiempo:new Date(),
    etiqueta:"Estado estimado"
  });

  escribir("mensaje","Orden enviada: "+texto+". La próxima actualización completa será en 20 segundos.");
  escribir("estadoSistema","Orden enviada. Esperando actualización del ciclo de 20 segundos.");

  enviarConReintento(autoVal,manualVal,texto,1);
}

function enviarConReintento(autoVal,manualVal,texto,intento){
  var maxIntentos=5;
  
  var url="https://api.thingspeak.com/update.json?api_key="+writeAPIKey+
        "&field3="+estadoEstimadoCalefactor+
        "&field4="+autoVal+
        "&field5="+manualVal;

  pedirTexto(url,function(err,respuesta,status){
    var ok=false;

    try{
      var obj=JSON.parse(respuesta);
      if(obj&&obj.entry_id&&Number(obj.entry_id)>0){ok=true;}
    }catch(e){
      if(parseInt(respuesta)>0){ok=true;}
    }

    if(ok){
      escribir("mensaje","Orden registrada en ThingSpeak: "+texto+". Actualización sincronizada cada 20 segundos.");
      controlEnProceso=false;
      bloquearTodo(false);
      actualizarSwitches();

      setTimeout(function(){leerDatos();},INTERVALO_WEB);
    }else{
      if(intento<maxIntentos){
        escribir("mensaje","ThingSpeak respondió 0. Reintentando en 20 segundos...");
        setTimeout(function(){enviarConReintento(autoVal,manualVal,texto,intento+1);},INTERVALO_WEB);
      }else{
        escribir("mensaje","No se confirmó la escritura. Revise API Key o límite de ThingSpeak.");
        controlEnProceso=false;
        bloquearTodo(false);
        actualizarSwitches();
      }
    }
  });
}

function generarEstadoVisualParaGrafica(feeds){
  var puntos=[];
  var modoAuto=0;
  var modoManual=0;
  var temperatura=null;
  var estado=estadoEstimadoCalefactor;

  if(!feeds||feeds.length===0){return puntos;}

  for(var i=0;i<feeds.length;i++){
    var f=feeds[i];

    if(esNumero(f.field1)){temperatura=parseFloat(f.field1);}
    if(esNumero(f.field4)){modoAuto=normalizar(parseFloat(f.field4));}
    if(esNumero(f.field5)){modoManual=normalizar(parseFloat(f.field5));}

    /*
      CORRECCIÓN:
      La web usa Field4/Field5 como mando principal cada 20 s.
      Field3 ya no domina la visualización si llega tarde.
      Así se evita que la web muestre OFF mientras el calefactor físico ya está ON.
    */

    if(modoManual===1){
      estado=1;
    }else if(modoAuto===1){
      if(temperatura!==null){
        if(temperatura<=25){estado=1;}
        else if(temperatura>=40){estado=0;}
      }
    }else{
      estado=0;
    }

    puntos.push({
      valor:estado,
      tiempo:new Date(f.created_at),
      etiqueta:formatearFechaHora(f.created_at)
    });
  }

  estadoEstimadoCalefactor=estado;
  return puntos;
}

  }
}

function puntosCampo(feeds,field){
  var puntos=[];
  for(var i=0;i<feeds.length;i++){
    var v=feeds[i][field];
    if(esNumero(v)){
      puntos.push({
        valor:parseFloat(v),
        tiempo:new Date(feeds[i].created_at),
        etiqueta:formatearFechaHora(feeds[i].created_at)
      });
    }
  }

  if(field==="field1"){puntosCampoCacheTemp=puntos;}

  return puntos;
}

function ultimoPunto(puntos){
  if(!puntos||puntos.length===0)return null;
  return puntos[puntos.length-1];
}

function calcularIndiceTermico(feeds){
  var puntos=[];
  for(var i=0;i<feeds.length;i++){
    var T=feeds[i].field1,H=feeds[i].field2;
    if(esNumero(T)&&esNumero(H)){
      T=parseFloat(T);H=parseFloat(H);
      puntos.push({
        valor:T+0.05*H,
        tiempo:new Date(feeds[i].created_at),
        etiqueta:formatearFechaHora(feeds[i].created_at)
      });
    }
  }
  return puntos;
}

function calcularDewPoint(feeds){
  var puntos=[];
  for(var i=0;i<feeds.length;i++){
    var T=feeds[i].field1,H=feeds[i].field2;
    if(esNumero(T)&&esNumero(H)&&parseFloat(H)>0){
      T=parseFloat(T);H=parseFloat(H);
      var a=17.27,b=237.7;
      var g=Math.log(H/100)+(a*T)/(b+T);
      var Td=(b*g)/(a-g);
      puntos.push({
        valor:Td,
        tiempo:new Date(feeds[i].created_at),
        etiqueta:formatearFechaHora(feeds[i].created_at)
      });
    }
  }
  return puntos;
}

function calcularVPD(feeds){
  var puntos=[];
  for(var i=0;i<feeds.length;i++){
    var T=feeds[i].field1,H=feeds[i].field2;
    if(esNumero(T)&&esNumero(H)){
      T=parseFloat(T);H=parseFloat(H);
      var es=0.6108*Math.exp((17.27*T)/(T+237.3));
      var ea=es*(H/100);
      puntos.push({
        valor:es-ea,
        tiempo:new Date(feeds[i].created_at),
        etiqueta:formatearFechaHora(feeds[i].created_at)
      });
    }
  }
  return puntos;
}

function formatearFechaHora(fecha){
  var d=new Date(fecha);
  return d.toLocaleString("es-EC",{day:"2-digit",month:"2-digit",year:"2-digit",hour:"2-digit",minute:"2-digit"});
}

function esNumero(v){return v!==null&&v!==undefined&&v!==""&&!isNaN(parseFloat(v));}
function normalizar(v){if(v===null||isNaN(v))return 0;return Number(v)>=1?1:0;}

function dibujarLineaPuntos(id,puntos,titulo,color){
  var c=document.getElementById(id),ctx=c.getContext("2d");
  prepararCanvas(c,ctx);
  var w=c.offsetWidth,h=360,pL=65,pR=25,pT=45,pB=78;
  limpiarGrafica(ctx,w,h,titulo);

  if(!puntos||puntos.length<2){sinDatos(ctx,w,h);return;}

  var vals=puntos.map(function(p){return p.valor;});
  var min=Math.min.apply(null,vals),max=Math.max.apply(null,vals);
  var margen=(max-min)*0.15;if(margen===0)margen=1;
  min-=margen;max+=margen;

  dibujarEjesTiempo(ctx,w,h,pL,pR,pT,pB,min,max,puntos);
  trazarSerie(ctx,w,h,pL,pR,pT,pB,min,max,puntos,color);
}

function dibujarMultiLineaPuntos(id,series,titulo){
  var c=document.getElementById(id),ctx=c.getContext("2d");
  prepararCanvas(c,ctx);
  var w=c.offsetWidth,h=360,pL=65,pR=25,pT=45,pB=90;
  limpiarGrafica(ctx,w,h,titulo);

  var todos=[],base=null;

  for(var s=0;s<series.length;s++){
    if(series[s].puntos&&series[s].puntos.length>0){
      base=series[s].puntos;
      for(var i=0;i<series[s].puntos.length;i++){
        todos.push(series[s].puntos[i].valor);
      }
    }
  }

  if(!base||todos.length<2){sinDatos(ctx,w,h);return;}

  var min=Math.min.apply(null,todos),max=Math.max.apply(null,todos);
  var margen=(max-min)*0.15;if(margen===0)margen=1;
  min-=margen;max+=margen;

  dibujarEjesTiempo(ctx,w,h,pL,pR,pT,pB,min,max,base);

  for(var s2=0;s2<series.length;s2++){
    trazarSerie(ctx,w,h,pL,pR,pT,pB,min,max,series[s2].puntos,series[s2].color);
  }

  dibujarLeyenda(ctx,series,pL,pT,w);
}

function dibujarLineaConLimitesPuntos(id,puntos,titulo,color,limites){
  var c=document.getElementById(id),ctx=c.getContext("2d");
  prepararCanvas(c,ctx);
  var w=c.offsetWidth,h=360,pL=65,pR=25,pT=45,pB=90;
  limpiarGrafica(ctx,w,h,titulo);

  if(!puntos||puntos.length<2){sinDatos(ctx,w,h);return;}

  var vals=puntos.map(function(p){return p.valor;});

  for(var l=0;l<limites.length;l++){vals.push(limites[l].valor);}

  var min=Math.min.apply(null,vals),max=Math.max.apply(null,vals);
  var margen=(max-min)*0.25;if(margen===0)margen=1;
  min-=margen;max+=margen;

  dibujarEjesTiempo(ctx,w,h,pL,pR,pT,pB,min,max,puntos);

  for(var i=0;i<limites.length;i++){
    var y=h-pB-(limites[i].valor-min)*(h-pT-pB)/(max-min);
    ctx.save();
    ctx.strokeStyle=limites[i].color;
    ctx.setLineDash([8,6]);
    ctx.lineWidth=1.5;
    ctx.beginPath();ctx.moveTo(pL,y);ctx.lineTo(w-pR,y);ctx.stroke();
    ctx.setLineDash([]);
    ctx.fillStyle=limites[i].color;
    ctx.font="bold 12px Arial";
    ctx.textAlign="right";
    ctx.fillText(limites[i].texto,w-pR-5,y-6);
    ctx.restore();
  }

  trazarSerie(ctx,w,h,pL,pR,pT,pB,min,max,puntos,color);
  dibujarLeyenda(ctx,[{nombre:"VPD [kPa]",color:color}],pL,pT,w);
}

function dibujarEscalonPuntos(id,puntos,titulo,color){
  var c=document.getElementById(id),ctx=c.getContext("2d");
  prepararCanvas(c,ctx);
  var w=c.offsetWidth,h=360,pL=65,pR=25,pT=45,pB=78;

  if(!puntos||puntos.length<2){sinDatos(ctx,w,h);return;}

  ctx.lineWidth=4;
  ctx.lineJoin="round";
  ctx.lineCap="round";

function trazarSerie(ctx,w,h,pL,pR,pT,pB,min,max,puntos,color){
  if(!puntos||puntos.length<2)return;

  ctx.beginPath();
  ctx.strokeStyle=color;
  ctx.lineWidth=2.6;
  ctx.lineJoin="round";
  ctx.lineCap="round";

  for(var i=0;i<puntos.length;i++){
    var x=pL+i*(w-pL-pR)/(puntos.length-1);
    var y=h-pB-(puntos[i].valor-min)*(h-pT-pB)/(max-min);

    if(i===0){ctx.moveTo(x,y);}
    else{ctx.lineTo(x,y);}
  }

  ctx.stroke();

  for(var j=0;j<puntos.length;j++){
    var x2=pL+j*(w-pL-pR)/(puntos.length-1);
    var y2=h-pB-(puntos[j].valor-min)*(h-pT-pB)/(max-min);

    ctx.beginPath();
    ctx.fillStyle=color;
    ctx.strokeStyle="#ffffff";
    ctx.lineWidth=1.3;
    ctx.arc(x2,y2,3.7,0,2*Math.PI);
    ctx.fill();
    ctx.stroke();
  }
}

function dibujarLeyenda(ctx,series,pL,pT,w){

function dibujarEjesTiempo(ctx,w,h,pL,pR,pT,pB,min,max,puntos){
  var plotW=w-pL-pR;
  var plotH=h-pT-pB;

  ctx.strokeStyle="#e2e8f0";
  ctx.lineWidth=1;

  for(var i=0;i<=5;i++){
    var y=pT+i*plotH/5;

    ctx.beginPath();
    ctx.moveTo(pL,y);
    ctx.lineTo(w-pR,y);
    ctx.stroke();

    var valor=max-(max-min)*i/5;

    ctx.fillStyle="#475569";
    ctx.font="12px Arial";
    ctx.textAlign="right";
    ctx.fillText(valor.toFixed(1),pL-8,y+4);
  }

  for(var j=0;j<=4;j++){
    var x=pL+j*plotW/4;

    ctx.beginPath();
    ctx.moveTo(x,pT);
    ctx.lineTo(x,h-pB);
    ctx.stroke();

    var idx=Math.round(j*(puntos.length-1)/4);

    if(puntos[idx]){
      ctx.save();
      ctx.translate(x,h-pB+18);
      ctx.rotate(-0.45);
      ctx.fillStyle="#475569";
      ctx.font="11px Arial";
      ctx.textAlign="right";
      ctx.fillText(puntos[idx].etiqueta,0,0);
      ctx.restore();
    }
  }

  ctx.strokeStyle="#334155";
  ctx.lineWidth=1.4;

  ctx.beginPath();
  ctx.moveTo(pL,pT);
  ctx.lineTo(pL,h-pB);
  ctx.lineTo(w-pR,h-pB);
  ctx.stroke();

  ctx.fillStyle="#475569";
  ctx.font="12px Arial";
  ctx.textAlign="center";
  ctx.fillText("Fecha y hora",w/2,h-8);
}

function dibujarEjesXY(ctx,w,h,pL,pR,pT,pB,xMin,xMax,yMin,yMax){
  var plotW=w-pL-pR;
  var plotH=h-pT-pB;

  ctx.strokeStyle="#e2e8f0";
  ctx.lineWidth=1;

  for(var i=0;i<=5;i++){

    ctx.fillStyle="#475569";
    ctx.font="12px Arial";

    ctx.fillText(valorY.toFixed(0),pL-8,y+4);
  }

  for(var j=0;j<=5;j++){
    var x=pL+j*plotW/5;
    var valorX=xMin+(xMax-xMin)*j/5;

    ctx.beginPath();
    ctx.moveTo(x,pT);
    ctx.lineTo(x,h-pB);
    ctx.stroke();

    ctx.fillStyle="#475569";
    ctx.font="12px Arial";
    ctx.textAlign="center";
    ctx.fillText(valorX.toFixed(0),x,h-pB+18);

  ctx.lineWidth=1.4;

  ctx.beginPath();
  ctx.moveTo(pL,pT);
  ctx.lineTo(pL,h-pB);
  ctx.lineTo(w-pR,h-pB);
  ctx.stroke();
}

function pedirJSON(url,cb){
  pedirTexto(url,function(err,text,status){
    if(err){
      cb(true,null);
      return;
    }

    try{
      cb(false,JSON.parse(text));
    }catch(e){
      cb(true,null);
    }
  });
}

function pedirTexto(url,cb){
  var xhr=new XMLHttpRequest();

  xhr.open("GET",url,true);

  xhr.onreadystatechange=function(){
    if(xhr.readyState===4){
      cb(!(xhr.status>=200&&xhr.status<300),xhr.responseText,xhr.status);
    }
  };

  xhr.onerror=function(){
    cb(true,"Error de red o CORS",0);
  };

  xhr.send();
}

var CHANNEL_ID = "3407423";

function ejecutarSistema(){
  document.getElementById("estadoEjecucion").innerHTML =
    "Ejecutando visualización. React y MATLAB Analysis se activan automáticamente con cada nueva inserción de datos.";

  leerDatosErgonomicos();
}

function leerDatosErgonomicos(){
  var url = "https://api.thingspeak.com/channels/" +
    CHANNEL_ID +
    "/feeds.json?results=36";

  if(READ_API_KEY !== ""){
    url += "&api_key=" + READ_API_KEY;
  }

  url += "&t=" + new Date().getTime();

  fetch(url)
    .then(function(response){ return response.json(); })
    .then(function(data){
      if(!data || !data.feeds || data.feeds.length === 0){
        document.getElementById("estadoEjecucion").innerHTML =
          "No se encontraron datos válidos en ThingSpeak.";
        return;
      }

      procesarDatos(data.feeds);
    })
    .catch(function(){
      document.getElementById("estadoEjecucion").innerHTML =
        "No se pudo leer ThingSpeak. Verifica canal o API Key.";
    });
}

function procesarDatos(feeds){
  var data = [];

  for(var i=0;i<feeds.length;i++){
    var T = parseFloat(feeds[i].field1);
    var H = parseFloat(feeds[i].field2);

    if(!isNaN(T) && !isNaN(H)){
      H = Math.max(1,Math.min(100,H));
      data.push({T:T,H:H});
    }
  }

  if(data.length === 0){
    document.getElementById("estadoEjecucion").innerHTML =
      "Los campos de temperatura y humedad no tienen datos válidos.";
    return;
  }

  var last = data[data.length-1];
  var T = last.T;
  var H = last.H;

  var HI = heatIndexC(T,H);
  var Td = dewPointC(T,H);
  var Tw = wetBulbC(T,H);
  var WBGT = 0.7*Tw + 0.3*T;
  var DI = T - (0.55 - 0.0055*H)*(T - 14.5);

  var PMV = estimarPMV(T,H);
  var PPD = 100 - 95*Math.exp(-0.03353*Math.pow(PMV,4) - 0.2179*Math.pow(PMV,2));

  var exposure = 0;

  for(var j=0;j<data.length;j++){
    var hiTemp = heatIndexC(data[j].T,data[j].H);
    var twTemp = wetBulbC(data[j].T,data[j].H);
    var wbgtTemp = 0.7*twTemp + 0.3*data[j].T;

    if(hiTemp >= 27 || wbgtTemp >= 25){
      exposure++;
    }
  }

  var exposureMin = exposure * 5;
  var riskScore = calcularRiesgoDifuso(HI,WBGT,Td,DI,PMV,PPD,exposure/data.length);

  actualizarInterfaz(T,H,HI,WBGT,Td,PMV,PPD,exposureMin,riskScore);

  document.getElementById("resultados").style.display = "block";
  document.getElementById("estadoEjecucion").innerHTML =
    "Resultados actualizados correctamente.";
}

function actualizarInterfaz(T,H,HI,WBGT,Td,PMV,PPD,exposureMin,riskScore){

  document.getElementById("wbgt").innerHTML = WBGT.toFixed(1) + " °C";
  document.getElementById("dewPoint").innerHTML = Td.toFixed(1) + " °C";
  document.getElementById("pmv").innerHTML = PMV.toFixed(2);
  document.getElementById("ppd").innerHTML = PPD.toFixed(0) + " %";
  document.getElementById("exposure").innerHTML = exposureMin.toFixed(0) + " min";

  var text = "";
  var recommendation = "";
  var color = "";
  var soft = "";

  if(riskScore < 0.75){
    text = "CONFORTABLE";
    recommendation = "Mantener monitoreo periódico. Condición térmica aceptable.";
    color = "#16a34a";
    soft = "#f0fdf4";
  }else if(riskScore < 1.5){
    text = "VIGILANCIA";
    recommendation = "Verificar ventilación, hidratación y evolución térmica.";
    color = "#2563eb";
    soft = "#eff6ff";
  }else if(riskScore < 2.5){
    text = "PRECAUCIÓN";
    recommendation = "Mejorar ventilación, reducir exposición y programar pausas.";
    color = "#f59e0b";
    soft = "#fffbeb";
  }else if(riskScore < 3.5){
    text = "ALTO";
    recommendation = "Aplicar pausas, reducir carga física y activar enfriamiento.";
    color = "#f97316";
    soft = "#fff7ed";
  }else{
    text = "CRÍTICO";
    recommendation = "Detener exposición y aplicar medidas correctivas inmediatas.";
    color = "#dc2626";
    soft = "#fef2f2";
  }

  document.getElementById("riskLevel").innerHTML = text;
  document.getElementById("riskLevel").style.color = color;
  document.getElementById("riskScore").innerHTML =
    "Índice difuso: " + riskScore.toFixed(2) + " / 4";

    "linear-gradient(135deg,#ffffff," + soft + ")";

  var pos = Math.max(0,Math.min(100,(riskScore/4)*100));
  document.getElementById("riskMarker").style.left = "calc(" + pos + "% - 12px)";
}

function heatIndexC(Tc,RH){
  var Tf = Tc*9/5 + 32;
  RH = Math.max(1,Math.min(100,RH));

  var HI_simple = 0.5 * (Tf + 61.0 + ((Tf - 68.0)*1.2) + (RH*0.094));
  var HIf;

  if(HI_simple < 80){
    HIf = HI_simple;
  }else{
    HIf = -42.379 + 2.04901523*Tf + 10.14333127*RH
      - 0.22475541*Tf*RH - 0.00683783*Math.pow(Tf,2)
      - 0.05481717*Math.pow(RH,2)
      + 0.00122874*Math.pow(Tf,2)*RH
      + 0.00085282*Tf*Math.pow(RH,2)
      - 0.00000199*Math.pow(Tf,2)*Math.pow(RH,2);
  }

  return (HIf - 32)*5/9;
}

function dewPointC(T,RH){
  RH = Math.max(1,Math.min(100,RH));
  var a = 17.27;
  var b = 237.7;
  var gamma = (a*T)/(b+T) + Math.log(RH/100);
  return (b*gamma)/(a-gamma);
}

function wetBulbC(T,RH){
  RH = Math.max(1,Math.min(100,RH));

  return T*Math.atan(0.151977*Math.sqrt(RH + 8.313659))
    + Math.atan(T + RH)
    - Math.atan(RH - 1.676331)
    + 0.00391838*Math.pow(RH,1.5)*Math.atan(0.023101*RH)
    - 4.686035;
}

function estimarPMV(T,H){
  var base = (T - 24) / 4;
  var humedad = (H - 50) / 100;
  return base + humedad;
}

function calcularRiesgoDifuso(HI,WBGT,Td,DI,PMV,PPD,exposure){
  var score = 0;

  if(HI >= 27) score += 0.7;
  if(HI >= 32) score += 0.8;
  if(HI >= 40) score += 1.0;

  if(WBGT >= 25) score += 0.6;
  if(WBGT >= 28.5) score += 0.8;
  if(WBGT >= 32) score += 1.0;

  if(Td >= 21) score += 0.5;
  if(DI >= 27.5) score += 0.5;
  if(PMV >= 1.2) score += 0.5;
  if(PPD >= 40) score += 0.5;
  if(exposure >= 0.45) score += 0.5;
  if(exposure >= 0.75) score += 0.5;

  return Math.max(0,Math.min(4,score));
}
