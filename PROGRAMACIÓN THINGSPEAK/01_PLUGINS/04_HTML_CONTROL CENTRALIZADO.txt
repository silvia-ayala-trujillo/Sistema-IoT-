function leerEstacion(id){
  var e = estaciones[id];

  var url = "https://api.thingspeak.com/channels/" +
    e.channelID +
    "/feeds/last.json?api_key=" +
    e.readAPIKey +
    "&t=" + new Date().getTime();

  fetch(url)
    .then(function(response){
      return response.json();
    })
    .then(function(f){

      if(!f || f.field1 === undefined){
        marcarOffline(id);
        return;
      }

      var temp = parseFloat(f.field1);
      var hum = parseFloat(f.field2);
      var calefactor = parseInt(f.field3);
      var auto = parseInt(f.field4);
      var manual = parseInt(f.field5);

      actualizarVista(id,temp,hum,calefactor,auto,manual);
    })
    .catch(function(){
      marcarOffline(id);
    });
}

function actualizarVista(id,temp,hum,calefactor,auto,manual){
  var status = document.getElementById("status"+id);
  status.innerHTML = "EN LÍNEA";
  status.style.background = "#dcfce7";
  status.style.color = "#15803d";

  if(!isNaN(temp)){
    ultimoDatoValido[id].temp = temp;
    document.getElementById("temp"+id).innerHTML = temp.toFixed(1);
  }else if(ultimoDatoValido[id].temp !== null){
    document.getElementById("temp"+id).innerHTML =
      ultimoDatoValido[id].temp.toFixed(1);
  }

  if(!isNaN(hum)){
    ultimoDatoValido[id].hum = hum;
    document.getElementById("hum"+id).innerHTML = hum.toFixed(0);
  }else if(ultimoDatoValido[id].hum !== null){
    document.getElementById("hum"+id).innerHTML =
      ultimoDatoValido[id].hum.toFixed(0);
  }

  var cal = document.getElementById("cal"+id);
  var calTxt = document.getElementById("calTxt"+id);

  if(calefactor === 1 || calefactor === 0){
    ultimoDatoValido[id].calefactor = calefactor;
  }else{
    calefactor = ultimoDatoValido[id].calefactor;
  }

  if(calefactor === 1){
    cal.innerHTML = "ON";
    cal.className = "value on";
    calTxt.innerHTML = "Encendido";
  }else if(calefactor === 0){
    cal.innerHTML = "OFF";
    cal.className = "value off";
    calTxt.innerHTML = "Apagado";
  }

  var modo = document.getElementById("modo"+id);
  var modoTxt = document.getElementById("modoTxt"+id);

  if(auto === 1 || auto === 0){
    ultimoDatoValido[id].auto = auto;
  }else{
    auto = ultimoDatoValido[id].auto;
  }

  if(manual === 1 || manual === 0){
    ultimoDatoValido[id].manual = manual;
  }else{
    manual = ultimoDatoValido[id].manual;
  }

  if(auto === 1){
    modo.innerHTML = "AUTO";
    modo.className = "value auto";
    modoTxt.innerHTML = "Automático";
  }else if(manual === 1){
    modo.innerHTML = "MANUAL";
    modo.className = "value manual";
    modoTxt.innerHTML = "Manual";
  }else if(auto === 0 && manual === 0){
    modo.innerHTML = "OFF";
    modo.className = "value off";
    modoTxt.innerHTML = "Sin modo activo";
  }
}

function leerHistorico(){
  Promise.all([
    fetch("https://api.thingspeak.com/channels/" + estaciones[1].channelID + "/feeds.json?api_key=" + estaciones[1].readAPIKey + "&results=30&t=" + new Date().getTime()).then(function(r){return r.json();}),

    fetch("https://api.thingspeak.com/channels/" + estaciones[3].channelID + "/feeds.json?api_key=" + estaciones[3].readAPIKey + "&results=30&t=" + new Date().getTime()).then(function(r){return r.json();})
  ])
  .then(function(res){
    var data1 = res[0].feeds || [];
    var data2 = res[1].feeds || [];
    var data3 = res[2].feeds || [];

    dibujarGraficaTemperatura(data1,data2,data3);
    dibujarGraficaHumedad(data1,data2,data3);
  })
  .catch(function(error){

  });

function dibujarGraficaTemperatura(data1,data2,data3){
  dibujarGrafica("graficaTemp",data1,data2,data3,"field1",0,50,"#2563eb","#f97316","#7c3aed");
}

function dibujarGraficaHumedad(data1,data2,data3){
  dibujarGrafica("graficaHum",data1,data2,data3,"field2",0,100,"#2563eb","#f97316","#7c3aed");
}

function dibujarGrafica(canvasId,data1,data2,data3,field,minY,maxY,color1,color2,color3){
  var canvas = document.getElementById(canvasId);
  var ctx = canvas.getContext("2d");

  ctx.clearRect(0,0,canvas.width,canvas.height);

  var w = canvas.width;
  var h = canvas.height;
  var left = 45;
  var right = 20;
  var top = 20;
  var bottom = 35;

  ctx.strokeStyle = "#e2e8f0";
  ctx.lineWidth = 1;
  ctx.font = "12px Arial";
  ctx.fillStyle = "#475569";

  for(var i=0;i<=5;i++){
    var y = top + i * ((h - top - bottom) / 5);
    var valor = maxY - i * ((maxY - minY) / 5);

    ctx.beginPath();
    ctx.moveTo(left,y);
    ctx.lineTo(w-right,y);
    ctx.stroke();

    ctx.fillText(valor.toFixed(0),8,y+4);
  }

  ctx.strokeStyle = "#94a3b8";
  ctx.beginPath();
  ctx.moveTo(left,top);
  ctx.lineTo(left,h-bottom);
  ctx.lineTo(w-right,h-bottom);
  ctx.stroke();

  function puntoY(valor){
    return top + (maxY - valor) * ((h - top - bottom) / (maxY - minY));
  }

  function puntoX(index,total){
    if(total <= 1) return left;
    return left + index * ((w - left - right) / (total - 1));
  }

  function dibujarLinea(data,color){
    ctx.strokeStyle = color;
    ctx.lineWidth = 2;
    ctx.beginPath();

    var iniciado = false;

    for(var i=0;i<data.length;i++){
      var valor = parseFloat(data[i][field]);

      if(isNaN(valor)) continue;

      var x = puntoX(i,data.length);
      var y = puntoY(valor);

      if(!iniciado){
        ctx.moveTo(x,y);
        iniciado = true;
      }else{
        ctx.lineTo(x,y);
      }
    }

    ctx.stroke();
  }

  dibujarLinea(data1,color1);
  dibujarLinea(data2,color2);
  dibujarLinea(data3,color3);

  ctx.fillStyle = color1;
  ctx.fillRect(left, h-18, 18, 3);
  ctx.fillStyle = "#334155";
  ctx.fillText("Estación 1 - Maestro", left+25, h-14);

  ctx.fillStyle = color2;
  ctx.fillRect(left+190, h-18, 18, 3);
  ctx.fillStyle = "#334155";
  ctx.fillText("Estación 2 - Esclavo", left+215, h-14);

  ctx.fillStyle = color3;
  ctx.fillRect(left+380, h-18, 18, 3);
  ctx.fillStyle = "#334155";
  ctx.fillText("Estación 3 - Esclavo", left+405, h-14);
}
